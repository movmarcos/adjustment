-- COMBINED FACT
select 
    'dev' as env,
    m.cobid,
    b.book_code,
    SUM(PNL_VECTOR_VALUE_IN_USD)   AS TOTAL_PNL_VECTOR_VALUE_IN_USD 
from fact.var_measures_combined m
inner join dimension.book b on m.book_key = b.book_key
where cobid in (20260605,20260608)
and b.entity_code = 'MUSI'
and b.book_code = 'BOOK-GEF'
and b.department_code = 'EFE'
group by all
union all
select 
    'prod' as env,
    m.cobid,
    b.book_code,
    SUM(PNL_VECTOR_VALUE_IN_USD)   AS TOTAL_PNL_VECTOR_VALUE_IN_USD 
from prod_raptor.fact.var_measures_combined m
inner join dimension.book b on m.book_key = b.book_key
where cobid in (20260605,20260608)
and b.entity_code = 'MUSI'
and b.book_code = 'BOOK-GEF'
and b.department_code = 'EFE'
group by all;

  SELECT COUNT(*) AS rows_returned, COUNT(DISTINCT fact.VAR_SUBCOMPONENT_ID) AS subcomps
  FROM FACT.VAR_MEASURES_COMBINED fact
  WHERE fact.COBID = 20260605                              -- SOURCE cob
    AND fact.PNL_VECTOR_VALUE_IN_USD IS NOT NULL
    AND fact.ENTITY_CODE = 'MUSI'                          -- engine's join_cond on ENTITY_CODE
    AND EXISTS (SELECT 1 FROM DIMENSION.BOOK bk
                WHERE bk.BOOK_KEY = COALESCE(fact.BOOK_KEY, -1)
                  AND bk.BOOK_CODE = 'BOOK-GEF');          -- engine's BOOK EXISTS

  -- same query but WITHOUT the ENTITY_CODE line — does subcomps jump to 10?
  SELECT COUNT(DISTINCT fact.VAR_SUBCOMPONENT_ID) AS subcomps

    AND EXISTS (SELECT 1 FROM DIMENSION.BOOK bk
                WHERE bk.BOOK_KEY = COALESCE(fact.BOOK_KEY, -1)
