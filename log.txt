  WITH cte AS (
    SELECT fact.PNL_VECTOR_VALUE_IN_USD * adjust.SCALE_FACTOR_ADJUSTED AS v
    FROM FACT.VAR_MEASURES_COMBINED fact
    WHERE fact.COBID = adjust.SOURCE_COBID
      AND fact.PNL_VECTOR_VALUE_IN_USD IS NOT NULL
      AND EXISTS (SELECT 1 FROM DIMENSION.BOOK bk WHERE bk.BOOK_KEY = COALESCE(fact.BOOK_KEY,-1)
                  AND (bk.BOOK_CODE = adjust.BOOK_CODE OR adjust.BOOK_CODE IS NULL)
                  AND (bk.DEPARTMENT_CODE = adjust.DEPARTMENT_CODE OR adjust.DEPARTMENT_CODE IS NULL))
      AND (adjust.ENTITY_CODE = fact.ENTITY_CODE OR adjust.ENTITY_CODE IS NULL)
      AND fact.VAR_SUBCOMPONENT_ID = 7
    UNION ALL
    SELECT fact.PNL_VECTOR_VALUE_IN_USD * -1 AS v
    FROM FACT.VAR_MEASURES fact
    INNER JOIN ADJUSTMENT_APP.ADJ_HEADER adjust ON adjust.DIMENSION_ADJ_ID = 733288
    WHERE fact.COBID = adjust.COBID
      AND fact.PNL_VECTOR_VALUE_IN_USD IS NOT NULL
      AND EXISTS (SELECT 1 FROM DIMENSION.BOOK bk WHERE bk.BOOK_KEY = COALESCE(fact.BOOK_KEY,-1)
                  AND (bk.BOOK_CODE = adjust.BOOK_CODE OR adjust.BOOK_CODE IS NULL)
                  AND (bk.DEPARTMENT_CODE = adjust.DEPARTMENT_CODE OR adjust.DEPARTMENT_CODE IS NULL))
      AND (adjust.ENTITY_CODE = fact.ENTITY_CODE OR adjust.ENTITY_CODE IS NULL)
      AND fact.VAR_SUBCOMPONENT_ID = 7
  )
  SELECT SUM(v) AS cte_total, COUNT(*) AS rows FROM cte;
