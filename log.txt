SELECT ADJ_ID, DIMENSION_ADJ_ID, RUN_STATUS, ADJUSTMENT_ACTION,
         COBID, SOURCE_COBID, ENTITY_CODE, BOOK_CODE, DEPARTMENT_CODE,
         SCALE_FACTOR, SCALE_FACTOR_ADJUSTED, CREATED_DATE, RUN_LOG_ID
  FROM ADJUSTMENT_APP.ADJ_HEADER
  WHERE COBID = 20260608
    AND PROCESS_TYPE IN ('VaR','FRTBALL')
    AND IS_DELETED = FALSE
  ORDER BY CREATED_DATE;
