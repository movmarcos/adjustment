-- A) Is the DEPLOYED proc actually the instrumented version?
SELECT GET_DDL('PROCEDURE',
  'ADJUSTMENT_APP.SP_PROCESS_ADJUSTMENT(VARCHAR,VARCHAR,NUMBER)')
  ILIKE '%EROL_PROCESS_LOG%' AS instrumented;     -- expect TRUE

-- B) Does the table have the new columns (STATUS/STARTED_AT/ENDED_AT)?
DESC TABLE ADJUSTMENT_APP.EROL_PROCESS_LOG;

-- C) Any rows at all, ignoring your filters?
SELECT COUNT(*) AS n, MAX(LOGGED_AT) AS latest FROM ADJUSTMENT_APP.EROL_PROCESS_LOG;

-- D) Did an EntityRoll actually REACH processing? (status is the key)
SELECT ADJ_ID, RUN_STATUS, ENTITY_CODE, COBID, SOURCE_COBID,
       START_DATE, PROCESS_DATE, RUN_LOG_ID, ERRORMESSAGE
FROM ADJUSTMENT_APP.ADJ_HEADER
WHERE ADJUSTMENT_ACTION = 'EntityRoll'
ORDER BY CREATED_DATE DESC LIMIT 10;
