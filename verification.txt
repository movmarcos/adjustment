use DVLP_RAPTOR_NEWADJ_2;
-- Adjustment reconciliation query with COALESCE-based dimension filters
-- Co-authored with CoCo
WITH adj AS (
    SELECT
        h.COBID,
        h.DEPARTMENT_CODE,
        h.BOOK_CODE,
        h.TRADE_TYPOLOGY        AS TRADE_TYPE,
        h.TRADE_CODE,
        h.PROCESS_TYPE,
        h.ADJUSTMENT_TYPE,
        h.ENTITY_CODE,
        h.DIMENSION_ADJ_ID      AS DEV_ADJ_ID,
        h.REASON,
        h.SOURCE_COBID,
        h.SCALE_FACTOR
    FROM ADJUSTMENT_APP.ADJ_HEADER h
    WHERE h.RUN_STATUS = 'Processed'
      AND h.IS_DELETED = FALSE
      AND h.DIMENSION_ADJ_ID IS NOT NULL
      AND dayname(to_date(cast(h.SOURCE_COBID as string), 'yyyyMMdd')) not in ('Sat', 'Sun')
     -- and h.ADJUSTMENT_TYPE <> 'EROL'
),
adjustment_values AS (
    SELECT 'VaR' AS PROCESS_TYPE, ADJUSTMENT_ID, SUM(PNL_VECTOR_VALUE_IN_USD) AS ADJUSTMENT_VALUE
    FROM FACT.VAR_MEASURES_ADJUSTMENT
    WHERE ADJUSTMENT_ID IN (SELECT DEV_ADJ_ID FROM adj)
    GROUP BY ADJUSTMENT_ID

    UNION ALL

    SELECT 'Stress', ADJUSTMENT_ID, SUM(SIMULATION_PL_IN_USD)
    FROM FACT.STRESS_MEASURES_ADJUSTMENT
    WHERE ADJUSTMENT_ID IN (SELECT DEV_ADJ_ID FROM adj)
    GROUP BY ADJUSTMENT_ID

    UNION ALL

    SELECT 'Sensitivity', ADJUSTMENT_ID, SUM(MEASURE_VALUE_USD)
    FROM FACT.SENSITIVITY_MEASURES_ADJUSTMENT
    WHERE ADJUSTMENT_ID IN (SELECT DEV_ADJ_ID FROM adj)
    GROUP BY ADJUSTMENT_ID

    UNION ALL

    SELECT 'FRTB', ADJUSTMENT_ID, SUM(AMOUNT_IN_USD)
    FROM FACT.FRTBSA_SENSITIVITY_MEASURES_ADJUSTMENT
    WHERE ADJUSTMENT_ID IN (SELECT DEV_ADJ_ID FROM adj)
    GROUP BY ADJUSTMENT_ID

    UNION ALL

    SELECT 'FRTBDRC', ADJUSTMENT_ID, SUM(JTD_LOSS_USD)
    FROM FACT.FRTBSA_DRC_MEASURES_ADJUSTMENT
    WHERE ADJUSTMENT_ID IN (SELECT DEV_ADJ_ID FROM adj)
    GROUP BY ADJUSTMENT_ID

    UNION ALL

    SELECT 'FRTBRRAO', ADJUSTMENT_ID, SUM(NOTIONAL_AMOUNT_USD)
    FROM FACT.FRTBSA_RRAO_MEASURES_ADJUSTMENT
    WHERE ADJUSTMENT_ID IN (SELECT DEV_ADJ_ID FROM adj)
    GROUP BY ADJUSTMENT_ID
),
original_values AS (
    -- VaR: join via BOOK_KEY and TRADE_KEY to dimension tables
    SELECT 'VaR' AS PROCESS_TYPE, a.COBID, a.ENTITY_CODE, a.BOOK_CODE,
           a.DEPARTMENT_CODE, a.TRADE_TYPE AS TRADE_TYPOLOGY,
           SUM(v.PNL_VECTOR_VALUE_IN_USD) AS ORIGINAL_VALUE
    FROM FACT.VAR_MEASURES v
    INNER JOIN DIMENSION.BOOK b
           ON b.BOOK_KEY = v.BOOK_KEY 
    INNER JOIN DIMENSION.TRADE t
           ON t.TRADE_KEY = v.TRADE_KEY 
    INNER JOIN adj a
           ON a.PROCESS_TYPE = 'VaR'
          AND v.COBID = a.COBID
          AND b.ENTITY_CODE = a.ENTITY_CODE
          AND (t.TRADE_CODE = coalesce(a.TRADE_CODE,t.TRADE_CODE))
          AND (b.BOOK_CODE = coalesce(a.BOOK_CODE,b.BOOK_CODE))
          AND (b.DEPARTMENT_CODE = coalesce(a.DEPARTMENT_CODE,b.DEPARTMENT_CODE))
          AND (t.TRADE_TYPOLOGY = COALESCE(a.TRADE_TYPE,t.TRADE_TYPOLOGY))
    GROUP BY a.COBID, a.ENTITY_CODE, a.BOOK_CODE, a.DEPARTMENT_CODE, a.TRADE_TYPE

    UNION ALL

    -- Stress: join via BOOK_KEY and TRADE_KEY to dimension tables
    SELECT 'Stress', a.COBID, a.ENTITY_CODE, a.BOOK_CODE,
           a.DEPARTMENT_CODE, a.TRADE_TYPE,
           SUM(s.SIMULATION_PL_IN_USD) AS ORIGINAL_VALUE
    FROM FACT.STRESS_MEASURES s
    INNER JOIN DIMENSION.BOOK b
           ON b.BOOK_KEY = s.BOOK_KEY 
    INNER JOIN DIMENSION.TRADE t
           ON t.TRADE_KEY = s.TRADE_KEY 
    INNER JOIN adj a
           ON a.PROCESS_TYPE = 'Stress'
          AND s.COBID = a.COBID
          AND b.ENTITY_CODE = a.ENTITY_CODE
          AND (t.TRADE_CODE = coalesce(a.TRADE_CODE,t.TRADE_CODE))
          AND (b.BOOK_CODE = coalesce(a.BOOK_CODE,b.BOOK_CODE))
          AND (b.DEPARTMENT_CODE = coalesce(a.DEPARTMENT_CODE,b.DEPARTMENT_CODE))
          AND (t.TRADE_TYPOLOGY = COALESCE(a.TRADE_TYPE,t.TRADE_TYPOLOGY))
    GROUP BY a.COBID, a.ENTITY_CODE, a.BOOK_CODE, a.DEPARTMENT_CODE, a.TRADE_TYPE

    UNION ALL

    -- Sensitivity: join via BOOK_KEY and TRADE_KEY to dimension tables
    SELECT 'Sensitivity', a.COBID, a.ENTITY_CODE, a.BOOK_CODE,
           a.DEPARTMENT_CODE, a.TRADE_TYPE,
           SUM(se.MEASURE_VALUE_USD) AS ORIGINAL_VALUE
    FROM FACT.SENSITIVITY_MEASURES se
    INNER JOIN DIMENSION.BOOK b
           ON b.BOOK_KEY = se.BOOK_KEY 
    INNER JOIN DIMENSION.TRADE t
           ON t.TRADE_KEY = se.TRADE_KEY 
    INNER JOIN adj a
           ON a.PROCESS_TYPE = 'Sensitivity'
          AND se.COBID = a.COBID
          AND b.ENTITY_CODE = a.ENTITY_CODE
          AND (t.TRADE_CODE = coalesce(a.TRADE_CODE,t.TRADE_CODE))
          AND (b.BOOK_CODE = coalesce(a.BOOK_CODE,b.BOOK_CODE))
          AND (b.DEPARTMENT_CODE = coalesce(a.DEPARTMENT_CODE,b.DEPARTMENT_CODE))
          AND (t.TRADE_TYPOLOGY = COALESCE(a.TRADE_TYPE,t.TRADE_TYPOLOGY))
    GROUP BY a.COBID, a.ENTITY_CODE, a.BOOK_CODE, a.DEPARTMENT_CODE, a.TRADE_TYPE

    UNION ALL

    -- FRTB: has ENTITY_CODE, BOOK_CODE, TRADE_TYPOLOGY directly
    SELECT 'FRTB', a.COBID, a.ENTITY_CODE, a.BOOK_CODE,
           a.DEPARTMENT_CODE, a.TRADE_TYPE,
           SUM(f.AMOUNT_IN_USD) AS ORIGINAL_VALUE
    FROM FACT.FRTBSA_SENSITIVITY_MEASURES f
    INNER JOIN DIMENSION.BOOK b
           ON b.BOOK_KEY = f.BOOK_KEY 
    INNER JOIN DIMENSION.TRADE t
           ON t.TRADE_KEY = f.TRADE_KEY 
    INNER JOIN adj a
           ON a.PROCESS_TYPE = 'FRTB'
          AND f.COBID = a.COBID
          AND f.ENTITY_CODE = a.ENTITY_CODE
          AND (t.TRADE_CODE = coalesce(a.TRADE_CODE,t.TRADE_CODE))
          AND (b.BOOK_CODE = coalesce(a.BOOK_CODE,b.BOOK_CODE))
          AND (b.DEPARTMENT_CODE = coalesce(a.DEPARTMENT_CODE,b.DEPARTMENT_CODE))
          AND (t.TRADE_TYPOLOGY = COALESCE(a.TRADE_TYPE, t.TRADE_TYPOLOGY))
    GROUP BY a.COBID, a.ENTITY_CODE, a.BOOK_CODE, a.DEPARTMENT_CODE, a.TRADE_TYPE

    UNION ALL

    -- FRTB: has ENTITY_CODE, BOOK_CODE, TRADE_TYPOLOGY directly
    SELECT 'FRTBDRC', a.COBID, a.ENTITY_CODE, a.BOOK_CODE,
           a.DEPARTMENT_CODE, a.TRADE_TYPE,
           SUM(f.JTD_LOSS_USD) AS ORIGINAL_VALUE
    FROM FACT.FRTBSA_DRC_MEASURES f
    INNER JOIN DIMENSION.BOOK b
           ON b.BOOK_KEY = f.BOOK_KEY 
    INNER JOIN DIMENSION.TRADE t
           ON t.TRADE_KEY = f.TRADE_KEY 
    INNER JOIN adj a
           ON a.PROCESS_TYPE = 'FRTBDRC'
          AND f.COBID = a.COBID
          AND f.ENTITY_CODE = a.ENTITY_CODE
          AND (t.TRADE_CODE = coalesce(a.TRADE_CODE,t.TRADE_CODE))
          AND (b.BOOK_CODE = coalesce(a.BOOK_CODE,b.BOOK_CODE))
          AND (b.DEPARTMENT_CODE = coalesce(a.DEPARTMENT_CODE,b.DEPARTMENT_CODE))
          AND (t.TRADE_TYPOLOGY = COALESCE(a.TRADE_TYPE, t.TRADE_TYPOLOGY))
    GROUP BY a.COBID, a.ENTITY_CODE, a.BOOK_CODE, a.DEPARTMENT_CODE, a.TRADE_TYPE

    UNION ALL

    -- FRTB: has ENTITY_CODE, BOOK_CODE, TRADE_TYPOLOGY directly
    SELECT 'FRTBRRAO', a.COBID, a.ENTITY_CODE, a.BOOK_CODE,
           a.DEPARTMENT_CODE, a.TRADE_TYPE,
           SUM(f.NOTIONAL_AMOUNT_USD) AS ORIGINAL_VALUE
    FROM FACT.FRTBSA_RRAO_MEASURES f
    INNER JOIN DIMENSION.BOOK b
           ON b.BOOK_KEY = f.BOOK_KEY 
    INNER JOIN DIMENSION.TRADE t
           ON t.TRADE_KEY = f.TRADE_KEY 
    INNER JOIN adj a
           ON a.PROCESS_TYPE = 'FRTBRRAO'
          AND f.COBID = a.COBID
          AND f.ENTITY_CODE = a.ENTITY_CODE
          AND (t.TRADE_CODE = coalesce(a.TRADE_CODE,t.TRADE_CODE))
          AND (b.BOOK_CODE = coalesce(a.BOOK_CODE,b.BOOK_CODE))
          AND (b.DEPARTMENT_CODE = coalesce(a.DEPARTMENT_CODE,b.DEPARTMENT_CODE))
          AND (t.TRADE_TYPOLOGY = COALESCE(a.TRADE_TYPE, t.TRADE_TYPOLOGY))
    GROUP BY a.COBID, a.ENTITY_CODE, a.BOOK_CODE, a.DEPARTMENT_CODE, a.TRADE_TYPE
),
-- For Roll adjustments: original value from the SOURCE_COBID (the date the roll came from)
roll_original_values AS (
    -- VaR at source COB
    SELECT 'VaR' AS PROCESS_TYPE, a.SOURCE_COBID, a.ENTITY_CODE, a.BOOK_CODE,
           a.DEPARTMENT_CODE, a.TRADE_TYPE AS TRADE_TYPOLOGY,
           SUM(v.PNL_VECTOR_VALUE_IN_USD) AS ROLL_SOURCE_VALUE
    FROM FACT.VAR_MEASURES_COMBINED v
    INNER JOIN DIMENSION.BOOK b
           ON b.BOOK_KEY = v.BOOK_KEY 
    INNER JOIN DIMENSION.TRADE t
           ON t.TRADE_KEY = v.TRADE_KEY 
    INNER JOIN adj a
           ON a.ADJUSTMENT_TYPE = 'Roll'
          AND a.PROCESS_TYPE = 'VaR'
          AND v.COBID = a.SOURCE_COBID
          AND b.ENTITY_CODE = a.ENTITY_CODE
          AND (t.TRADE_CODE = coalesce(a.TRADE_CODE,t.TRADE_CODE))
          AND (b.BOOK_CODE = coalesce(a.BOOK_CODE,b.BOOK_CODE))
          AND (b.DEPARTMENT_CODE = coalesce(a.DEPARTMENT_CODE,b.DEPARTMENT_CODE))
          AND (t.TRADE_TYPOLOGY = COALESCE(a.TRADE_TYPE,t.TRADE_TYPOLOGY))
    GROUP BY a.SOURCE_COBID, a.ENTITY_CODE, a.BOOK_CODE, a.DEPARTMENT_CODE, a.TRADE_TYPE

    UNION ALL

    -- Stress at source COB
    SELECT 'Stress', a.SOURCE_COBID, a.ENTITY_CODE, a.BOOK_CODE,
           a.DEPARTMENT_CODE, a.TRADE_TYPE,
           SUM(s.SIMULATION_PL_IN_USD)
    FROM FACT.STRESS_MEASURES_COMBINED s
    INNER JOIN DIMENSION.BOOK b
           ON b.BOOK_KEY = s.BOOK_KEY 
    INNER JOIN DIMENSION.TRADE t
           ON t.TRADE_KEY = s.TRADE_KEY 
    INNER JOIN adj a
           ON a.ADJUSTMENT_TYPE = 'Roll'
          AND a.PROCESS_TYPE = 'Stress'
          AND s.COBID = a.SOURCE_COBID
          AND b.ENTITY_CODE = a.ENTITY_CODE
          AND (t.TRADE_CODE = coalesce(a.TRADE_CODE,t.TRADE_CODE))
          AND (b.BOOK_CODE = coalesce(a.BOOK_CODE,b.BOOK_CODE))
          AND (b.DEPARTMENT_CODE = coalesce(a.DEPARTMENT_CODE,b.DEPARTMENT_CODE))
          AND (t.TRADE_TYPOLOGY = COALESCE(a.TRADE_TYPE,t.TRADE_TYPOLOGY))
    GROUP BY a.SOURCE_COBID, a.ENTITY_CODE, a.BOOK_CODE, a.DEPARTMENT_CODE, a.TRADE_TYPE

    UNION ALL

    -- Sensitivity at source COB
    SELECT 'Sensitivity', a.SOURCE_COBID, a.ENTITY_CODE, a.BOOK_CODE,
           a.DEPARTMENT_CODE, a.TRADE_TYPE,
           SUM(se.MEASURE_VALUE_USD)
    FROM FACT.SENSITIVITY_MEASURES_ADJUSTED se
    INNER JOIN DIMENSION.BOOK b
           ON b.BOOK_KEY = se.BOOK_KEY 
    INNER JOIN DIMENSION.TRADE t
           ON t.TRADE_KEY = se.TRADE_KEY
    INNER JOIN adj a
           ON a.ADJUSTMENT_TYPE = 'Roll'
          AND a.PROCESS_TYPE = 'Sensitivity'
          AND se.COBID = a.SOURCE_COBID
          AND b.ENTITY_CODE = a.ENTITY_CODE
          AND (t.TRADE_CODE = coalesce(a.TRADE_CODE,t.TRADE_CODE))
          AND (b.BOOK_CODE = coalesce(a.BOOK_CODE,b.BOOK_CODE))
          AND (b.DEPARTMENT_CODE = coalesce(a.DEPARTMENT_CODE,b.DEPARTMENT_CODE))
          AND (t.TRADE_TYPOLOGY = COALESCE(a.TRADE_TYPE,t.TRADE_TYPOLOGY))
    GROUP BY a.SOURCE_COBID, a.ENTITY_CODE, a.BOOK_CODE, a.DEPARTMENT_CODE, a.TRADE_TYPE

    UNION ALL

    -- FRTB at source COB
    SELECT 'FRTB', a.SOURCE_COBID, a.ENTITY_CODE, a.BOOK_CODE,
           a.DEPARTMENT_CODE, a.TRADE_TYPE,
           SUM(f.AMOUNT_IN_USD)
    FROM FACT.FRTBSA_SENSITIVITY_MEASURES_ADJUSTED f
    INNER JOIN DIMENSION.BOOK b
           ON b.BOOK_KEY = f.BOOK_KEY 
    INNER JOIN DIMENSION.TRADE t
           ON t.TRADE_KEY = f.TRADE_KEY
    INNER JOIN adj a
           ON a.ADJUSTMENT_TYPE = 'Roll'
          AND a.PROCESS_TYPE = 'FRTB'
          AND f.COBID = a.SOURCE_COBID
          AND f.ENTITY_CODE = a.ENTITY_CODE
          AND (t.TRADE_CODE = coalesce(a.TRADE_CODE,t.TRADE_CODE))
          AND (b.BOOK_CODE = coalesce(a.BOOK_CODE,b.BOOK_CODE))
          AND (b.DEPARTMENT_CODE = coalesce(a.DEPARTMENT_CODE,b.DEPARTMENT_CODE))
          AND (t.TRADE_TYPOLOGY = COALESCE(a.TRADE_TYPE,t.TRADE_TYPOLOGY))
    GROUP BY a.SOURCE_COBID, a.ENTITY_CODE, a.BOOK_CODE, a.DEPARTMENT_CODE, a.TRADE_TYPE

    UNION ALL

    -- FRTB at source COB
    SELECT 'FRTBDRC', a.SOURCE_COBID, a.ENTITY_CODE, a.BOOK_CODE,
           a.DEPARTMENT_CODE, a.TRADE_TYPE,
           SUM(f.JTD_LOSS_USD)
    FROM FACT.FRTBSA_DRC_MEASURES_ADJUSTED f
    INNER JOIN DIMENSION.BOOK b
           ON b.BOOK_KEY = f.BOOK_KEY 
    INNER JOIN DIMENSION.TRADE t
           ON t.TRADE_KEY = f.TRADE_KEY
    INNER JOIN adj a
           ON a.ADJUSTMENT_TYPE = 'Roll'
          AND a.PROCESS_TYPE = 'FRTBDRC'
          AND f.COBID = a.SOURCE_COBID
          AND f.ENTITY_CODE = a.ENTITY_CODE
          AND (t.TRADE_CODE = coalesce(a.TRADE_CODE,t.TRADE_CODE))
          AND (b.BOOK_CODE = coalesce(a.BOOK_CODE,b.BOOK_CODE))
          AND (b.DEPARTMENT_CODE = coalesce(a.DEPARTMENT_CODE,b.DEPARTMENT_CODE))
          AND (t.TRADE_TYPOLOGY = COALESCE(a.TRADE_TYPE,t.TRADE_TYPOLOGY))
    GROUP BY a.SOURCE_COBID, a.ENTITY_CODE, a.BOOK_CODE, a.DEPARTMENT_CODE, a.TRADE_TYPE

    UNION ALL

    -- FRTB at source COB
    SELECT 'FRTBRRAO', a.SOURCE_COBID, a.ENTITY_CODE, a.BOOK_CODE,
           a.DEPARTMENT_CODE, a.TRADE_TYPE,
           SUM(f.NOTIONAL_AMOUNT_USD)
    FROM FACT.FRTBSA_RRAO_MEASURES_ADJUSTED f
    INNER JOIN DIMENSION.BOOK b
           ON b.BOOK_KEY = f.BOOK_KEY 
    INNER JOIN DIMENSION.TRADE t
           ON t.TRADE_KEY = f.TRADE_KEY
    INNER JOIN adj a
           ON a.ADJUSTMENT_TYPE = 'Roll'
          AND a.PROCESS_TYPE = 'FRTBRRAO'
          AND f.COBID = a.SOURCE_COBID
          AND f.ENTITY_CODE = a.ENTITY_CODE
          AND (t.TRADE_CODE = coalesce(a.TRADE_CODE,t.TRADE_CODE))
          AND (b.BOOK_CODE = coalesce(a.BOOK_CODE,b.BOOK_CODE))
          AND (b.DEPARTMENT_CODE = coalesce(a.DEPARTMENT_CODE,b.DEPARTMENT_CODE))
          AND (t.TRADE_TYPOLOGY = COALESCE(a.TRADE_TYPE,t.TRADE_TYPOLOGY))
    GROUP BY a.SOURCE_COBID, a.ENTITY_CODE, a.BOOK_CODE, a.DEPARTMENT_CODE, a.TRADE_TYPE
)
SELECT
    a.DEV_ADJ_ID,
    a.COBID,
    a.ENTITY_CODE,
    a.PROCESS_TYPE,
    a.TRADE_CODE,
    a.DEPARTMENT_CODE,
    a.BOOK_CODE,
    a.TRADE_TYPE,
    a.ADJUSTMENT_TYPE,
    a.SCALE_FACTOR,
    av.ADJUSTMENT_VALUE,
    o.ORIGINAL_VALUE,
    a.SOURCE_COBID,
    r.ROLL_SOURCE_VALUE,
    a.REASON,
    -- Expected value per adjustment type
    CASE a.ADJUSTMENT_TYPE
        WHEN 'Roll'    THEN av.ADJUSTMENT_VALUE + o.ORIGINAL_VALUE
        WHEN 'Flatten' THEN o.ORIGINAL_VALUE * -1
        WHEN 'Scale'   THEN o.ORIGINAL_VALUE * a.SCALE_FACTOR
    END AS EXPECTED_VALUE,
    -- Difference (should be 0 if reconciled)
    CASE a.ADJUSTMENT_TYPE
        WHEN 'Roll'    THEN r.ROLL_SOURCE_VALUE - (av.ADJUSTMENT_VALUE + o.ORIGINAL_VALUE)
        WHEN 'Flatten' THEN av.ADJUSTMENT_VALUE - (o.ORIGINAL_VALUE * -1)
        WHEN 'Scale'   THEN av.ADJUSTMENT_VALUE + (o.ORIGINAL_VALUE * a.SCALE_FACTOR)
    END AS RECON_DIFFERENCE,
    -- Reconciliation status
    CASE
        WHEN av.ADJUSTMENT_VALUE IS NULL THEN 'MISSING ADJUSTMENT'
        WHEN o.ORIGINAL_VALUE IS NULL AND a.ADJUSTMENT_TYPE IN ('Flatten', 'Scale') THEN 'MISSING ORIGINAL'
        WHEN r.ROLL_SOURCE_VALUE IS NULL AND a.ADJUSTMENT_TYPE = 'Roll' THEN 'MISSING ROLL SOURCE'
        WHEN a.ADJUSTMENT_TYPE = 'Roll' AND ABS(r.ROLL_SOURCE_VALUE - (av.ADJUSTMENT_VALUE + ifnull(o.ORIGINAL_VALUE,0))) < 0.01 THEN 'RECONCILED'
        WHEN a.ADJUSTMENT_TYPE = 'Flatten' AND ABS(av.ADJUSTMENT_VALUE - (o.ORIGINAL_VALUE * -1)) < 0.01 THEN 'RECONCILED'
        WHEN a.ADJUSTMENT_TYPE = 'Scale' AND ABS(av.ADJUSTMENT_VALUE + (o.ORIGINAL_VALUE * a.SCALE_FACTOR)) < 0.01 THEN 'RECONCILED'
        ELSE 'BREAK'
    END AS RECON_STATUS
FROM adj a
LEFT JOIN adjustment_values av
       ON av.PROCESS_TYPE = a.PROCESS_TYPE
      AND av.ADJUSTMENT_ID = a.DEV_ADJ_ID
LEFT JOIN original_values o
       ON o.PROCESS_TYPE    = a.PROCESS_TYPE
      AND o.COBID           = a.COBID
      AND o.ENTITY_CODE     = a.ENTITY_CODE
      AND o.BOOK_CODE       IS NOT DISTINCT FROM a.BOOK_CODE
      AND o.DEPARTMENT_CODE IS NOT DISTINCT FROM a.DEPARTMENT_CODE
      AND o.TRADE_TYPOLOGY  IS NOT DISTINCT FROM a.TRADE_TYPE
LEFT JOIN roll_original_values r
       ON r.PROCESS_TYPE    = a.PROCESS_TYPE
      AND r.SOURCE_COBID    = a.SOURCE_COBID
      AND r.ENTITY_CODE     = a.ENTITY_CODE
      AND r.BOOK_CODE       IS NOT DISTINCT FROM a.BOOK_CODE
      AND r.DEPARTMENT_CODE IS NOT DISTINCT FROM a.DEPARTMENT_CODE
      AND r.TRADE_TYPOLOGY  IS NOT DISTINCT FROM a.TRADE_TYPE
      AND a.ADJUSTMENT_TYPE = 'Roll'
ORDER BY a.COBID, a.DEPARTMENT_CODE, a.BOOK_CODE;
